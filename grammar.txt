//Syntax of apg file (which specifies BNF-like).

apg_file: item* EOF

item: nonterminal

nonterminal: IDENT ':' expression* ';'

expression: expression '|' expression
| '(' expression ')' rep?
| (pred? expression)+
| primary rep?

rep: '?' | '*' | '+'

pred: ('&' | '!') expression

primary: IDENT
| terminal

terminal: quoted | range | EOF

quoted: '\'' [^\']* '\''

range: '[' '^'? (.-. | .)+ ']'
